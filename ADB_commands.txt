ADB commands:

adb   			//check
adb devices -l		//check whether the devices is connected 
// adb service
adb start-service
adb kill service
adb install ~/Desktop/netdroid.apk        //install the android application to the phone
adb shell pm list packages netdroid       // using pm(package manager) to list the full name(like org.pctechtips.netdroid) of the pakage you want to uninstall, otherwise you can't do it successfully
adb uninstall fullnameofthePackage	//you don't need to provide the path
adb shell pm list packages		//list all the packages
adb shell pm list packages | grep music	//列出含music的packages
adb shell pm path org.pctechtips.netdroid	//find path of specific package
adb shell install -r "/path/to/the/package"		//-r means uninstall
adb shell install -r "/path/to/the/package" apk:Desktop/netdroid.apk    //reinstall
adb shell am start -n  org.pctechtips.netdroid/org.pctechtips.netdroid.MainActivity      //am(app manager), start the app
adb shell am force-stop org.pctechtips.netdroid --user all		// stop app, using user mode
adb shell dumpsys window windows | grep -i focus	//see the current open app

adb shell ps      //list all the process
adb shell ps | grep netdroid		//see the process imformation(ID number)
adb shell kill -9 ps_ID   		//-9 be a group member	
adb ip addr show		// a list of all the IP address
adb shell ip addr show wlan0	//get information of a specific ID(wire less area network)
adb shell route		
adb shell dumpsys 		//give a lot of information about system
adb shell dumpsys > Desktop/dumpsys_info.txt			//save the information in specific file
adb shell dumpsys | less
adb shell dumpsys battery / cpuinfo / display / lacation/     //info about battery
adb shell getprop | grep hostname      //getprop(property)
adb shell getprop | grep o.build.version.sdk		//get the prop of the sdk version

adb backup -apk(all the app packages) -shared(all the sdcard content) -....(things you want to backup) -f Desktop/af/afas(file you want to save)

adb reboot
